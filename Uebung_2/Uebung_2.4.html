Mein CSS-Code:

        
            .content
            {
                font-family: Roboto;
                background-color: #a9d7d1;
            }

            h1
            {
                margin-top: 0;
                text-align: center;
            }

            .elem 
            {
                background-color: snow;
                width: 100vh;
                height: 65vh ; /*100 viewport height - etwas einfacher als Prozentangabe, die hier nichts anzeigt*/
                margin: 0 auto; /*horizontales Zentrieren*/
                border-radius: 15px 15px 0 0;   
                /*
                hiermit rundet man die Ecken oben rechts und oben links ab. 
                Für welche Ecke stehen die Werte? Reihenfolge : oben links, oben rechts, unten rechts, unten links.
                */
                padding-top: 35px;
            }

            .headertext
            {
                text-align: center;
            }

            .columnLeft
            {
                display: inline-block;
                text-align: right;
                vertical-align: top;
                margin-top: 10px;
                margin-right: 15px;
                width: 60%;
            }

            .row
            {
                margin-top: 15px;
            }

            .columnRight
            {
                display: inline-block;
                text-align: left;
                margin-top: 10px;
                width: 30%;
            }

            .selectorItem
            {
                margin-top:20px
            }
